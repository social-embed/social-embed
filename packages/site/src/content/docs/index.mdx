      <a href="/lib"><code>@social-embed/lib</code></a> is a TypeScript library for:
    </p>
    <ul class="list-disc ml-5">
      <li>Extracting IDs from any recognized provider URL (YouTube, DailyMotion, Loom, and more).</li>
      <li>Converting those IDs into embeddable formats.</li>
      <li>Normalizing or validating incoming URLs.</li>
    </ul>

    <Code
      lang="ts"
      wrap
      code={`import {
    getProviderFromUrl,
    getYouTubeIdFromUrl,
    youTubeUrlRegex,
} from "@social-embed/lib";

"https://youtu.be/watch?v=Bd8_vO5zrjo".match(
    youTubeUrlRegex
);
// => [
//   "https://youtu.be/watch?v=Bd8_vO5zrjo","Bd8_vO5zrjo"
// ]

getYouTubeIdFromUrl(
    "https://youtu.be/watch?v=Bd8_vO5zrjo"
);
// => "Bd8_vO5zrjo"

const provider = getProviderFromUrl(
    "https://youtu.be/watch?v=Bd8_vO5zrjo"
);
// => {
//   name: "YouTube",
//   canParseUrl: [Function],
//   getIdFromUrl: [Function],
//   getEmbedUrlFromId: [Function]
// }

provider.getIdFromUrl(
    "https://youtu.be/watch?v=Bd8_vO5zrjo"
);
// => "Bd8_vO5zrjo"

provider.getEmbedUrlFromId("Bd8_vO5zrjo");
// => "https://www.youtube.com/embed/Bd8_vO5zrjo"
`}
/>
